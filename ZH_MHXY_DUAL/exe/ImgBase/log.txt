//取字符串长度，中文算2个字符
function getstrlength(str){
	var chineseregex = /[^\x00-\xff]/g;
	var strlength = str.replace(chineseregex,"**").length;
	return strlength;
}

//汉字校验
function ischinese(str)
{
	var reg=/^[\u4e00-\u9fa5]+$/i;
	if(reg.test(str))
	{//中文校验成功
		return true;
	}else
	{
		return false;
	}
}
//英文校验
function isalphabet(str)
{
	var reg=/^[abcdefghijklmnopqrstuvwxyzabcdefghijklmnopqrstuvwxyz]+$/i;
	if(reg.test(str))
	{//英文校验成功
		return true;
	}
	else
	{
		return false;
	}
}

function count(){
	if(countdowntime == 0){
		clearinterval(intervalid);
		getsmsbtn.show();
		countdowntip.attr('innerhtml','');
		sendsmstip.attr('innerhtml','');
		countdowntime = time;
		return;
	}
	countdowntip.attr('innerhtml','还有<span style="font-size:14px; font-weight:bold; margin:0 3px; color:red;">' + countdowntime + '</span>秒可以再次获取验证码');
	countdowntime = countdowntime - 1;
}

	var tip = '';

		if(arr[1].indexof("实名认证") !== -1){

		$("#feesmslist").show().html(htmlstr);

	// 2013-11-15 add by zhouqing -- begin
	else if(msg.indexof('202') !== -1){

$("#hassendbtn").live("click", function(){

	var $ischecking = $("#ischecking"),
	$hassendbtn = $(this),
	$ischeckederr = $("#ischeckederr"),
	$ischeckedok = $("#ischeckedok");

	var lazyarr = [1000, 2000, 3000],
	lazyidx = 0;

	var href = window.location.href,
	protocol = "http";
	if(href.indexof("https") == 0){protocol = "https";}

	(function docheck(){

		$hassendbtn.hide();
		$ischeckederr.hide();
		$ischecking.show();

		$.getjson(
			protocol+"://reg.163.com/interfaces/urs/verifyupsms.do?type=mibao2_replacemobile",
			//"/ftl/testdata2/reg/hassendsms.txt",
			function(data){
				$ischecking.hide();

				if(data.retcode-0 === 200){
					if(data.verifystatus-0 === 1){
						$ischeckedok.show();
						$("#mobcheckcode").val("000000");
						$("#upsms").removeattr("disabled");
						$("input[type='submit']").trigger("click");
					}else{
						if(lazyidx >= lazyarr.length){
							if(data.verifystatus-0 === -1){
								$ischeckederr.show().text("验证失败。短信错误，请确认已经正确编辑短信内容。");
							}else{
								$ischeckederr.show().text("验证失败。系统未收到短信，请确认已经正确发送。");
							}
							$hassendbtn.show();
							lazyidx = 0;
						}else{
							$ischecking.show();
							settimeout(docheck, lazyarr[lazyidx++]);
						}
					}
				}else{
					if(lazyidx >= lazyarr.length){
						if(data.retcode-0 === 453){
							$ischeckederr.show().text("用户未绑定手机");
						}else if(data.retcode-0 === 412){
							$ischeckederr.show().text("验证失败。您今天尝试验证的次数过多，请改天再试。");
						}else if(data.retcode-0 === 500){
							$ischeckederr.show().text("系统异常");
						}else if(data.retcode-0 === 401){
							$ischeckederr.show().text("传递的手机号错误");
						}
						$hassendbtn.show();
						lazyidx = 0;
					}else{
						$ischecking.show();
						settimeout(docheck, lazyarr[lazyidx++]);
					}
				}
			}
		);
	})();
});
